# ACP Migration Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Use superpowers:test-driven-development for every task. Use superpowers:verification-before-completion before claiming a task done.

**Goal:** Make specifyr speak the Agent Client Protocol (ACP) on both sides — as a server (so external UIs like AionUi/Zed can drive runs) and as a client (so any ACP-speaking coding agent can replace the stdout-parsing in `hermes-cli.js`).

**Architecture:**
- Add a thin `src/acp/` layer that owns the protocol wire format and translates between specifyr's existing event shapes and ACP's `session/update` notifications. The orchestration core ([orchestrator.js](src/core/orchestrator.js), [run-scheduler.js](src/core/run-scheduler.js), [turn-broker.js](src/core/turn-broker.js), [capability-approval-service.js](src/core/capability-approval-service.js)) does **not** change semantically.
- **Client side:** A new `AcpRunner` ([src/runners/acp.js](src/runners/acp.js)) replaces stdout parsing. It spawns an ACP-speaking agent over stdio, negotiates capabilities, runs `session/prompt`, and translates incoming `session/update` notifications into the existing `onEvent({type: "assistant", message: ...})` shape that [TurnBroker](src/core/turn-broker.js) and [HermesStreamingRunner](src/runners/hermes-streaming.js) already expect — drop-in compatible.
- **Server side:** A new stdio entrypoint `bin/specifyr-acp` ([bin/specifyr-acp.js](bin/specifyr-acp.js)) implements `Agent` from the SDK. It maps `session/new`/`session/load` to specifyr's `(slug, stepId, sid)` triplet and bridges [TurnBroker](src/core/turn-broker.js) emitter events back out as `session/update` notifications. `session/request_permission` is wired through [capability-approval-service.js](src/core/capability-approval-service.js) via a new `AcpApprovalTransport`.
- **Approvals** at *spec/plan/task_batch* level remain specifyr-internal (out of ACP's scope). Only *capability-level* approvals (filesystem write, secrets, network, payments) ride on `session/request_permission`.

**Tech Stack:**
- `@agentclientprotocol/sdk` (latest: 0.21.0, Apache-2.0, JSON-RPC over stdio + zod schemas)
- Node 22+ (project requires it)
- `node --test` for unit + integration tests
- Existing: chokidar, yaml, vue/nuxt (UI untouched in phases 0–4)

// src/acp/session-id.js
const SCHEME = "specifyr";
const VERSION = "1";
const SEP = ":";

function assertSafe(value, field) {
  if (typeof value !== "string" || value.length === 0) {
    throw new Error(`${field} must be a non-empty string`);
  }
  if (value.includes(SEP)) {
    throw new Error(`${field} must not contain the separator '${SEP}'`);
  }
}

export function encodeSessionId({ slug, stepId, sid }) {
  assertSafe(slug, "slug");
  assertSafe(stepId, "stepId");
  assertSafe(sid, "sid");
  return [SCHEME, VERSION, slug, stepId, sid].join(SEP);
}

export function decodeSessionId(id) {
  if (typeof id !== "string") throw new Error("malformed session-id: not a string");
  const parts = id.split(SEP);
  if (parts.length !== 5) throw new Error(`malformed session-id: expected 5 parts, got ${parts.length}`);
  const [scheme, version, slug, stepId, sid] = parts;
  if (scheme !== SCHEME) throw new Error(`unknown scheme '${scheme}'`);
  if (version !== VERSION) throw new Error(`unsupported version '${version}'`);
  return { slug, stepId, sid };
}
```

// src/acp/translate.js
/**
 * Translate one specifyr persisted event (as written by TurnBroker.append())
 * into one ACP SessionUpdate object — or null if the event has no ACP analogue.
 *
 * Caller wraps the result in:
 *   { sessionId, update: <returned> }
 * and emits it via Agent.sessionUpdate().
 *
 * @param {{event: string, data: any}} entry
 * @returns {object|null}
 */
export function specifyrEventToAcpUpdate({ event, data }) {
  switch (event) {
    case "claude":
      return translateClaudeEvent(data);
    case "session_reset":
      return {
        sessionUpdate: "plan",
        entries: [{ content: data?.message ?? "session reset", priority: "medium", status: "completed" }]
      };
    case "turn_failed":
      return {
        sessionUpdate: "agent_message_chunk",
        content: { type: "text", text: `\n\n[error] ${data?.message ?? "turn failed"}` }
      };
    case "assistant_message":
    case "done":
      // Already streamed via chunks. Caller signals stopReason on the prompt response.
      return null;
    default:
      return null;
  }
}

function translateClaudeEvent(data) {
  if (!data || typeof data !== "object") return null;
  if (data.type === "assistant" && Array.isArray(data.message?.content)) {
    // Walk content blocks. For a multi-block message we'd emit several updates;
    // caller currently only handles one — see Task 2.5 where bridging splits them.
    const block = data.message.content[0];
    if (!block) return null;
    if (block.type === "text") {
      return { sessionUpdate: "agent_message_chunk", content: { type: "text", text: block.text } };
    }
    if (block.type === "tool_use") {
      return {
        sessionUpdate: "tool_call",
        toolCallId: block.id,
        title: block.name,
        kind: inferToolKind(block.name),
        status: "in_progress",
        rawInput: block.input
      };
    }
  }
  if (data.type === "user" && Array.isArray(data.message?.content)) {
    const block = data.message.content[0];
    if (block?.type === "tool_result") {
      return {
        sessionUpdate: "tool_call_update",
        toolCallId: block.tool_use_id,
        status: block.is_error ? "failed" : "completed",
        content: typeof block.content === "string"
          ? [{ type: "content", content: { type: "text", text: block.content } }]
          : undefined
      };
    }
  }
  return null;
}

function inferToolKind(name) {
  const n = String(name ?? "").toLowerCase();
  if (n === "read" || n.includes("read")) return "read";
  if (n === "edit" || n === "write" || n.includes("write") || n.includes("edit")) return "edit";
  if (n === "bash" || n.includes("exec")) return "execute";
  if (n.includes("search") || n === "grep" || n === "glob") return "search";
  if (n.includes("fetch") || n.includes("http")) return "fetch";
  return "other";
}
```

// src/runners/acp.js
import { spawn } from "node:child_process";
import { Readable, Writable } from "node:stream";
import { ClientSideConnection } from "@agentclientprotocol/sdk";
import { acpUpdateToSpecifyrEvent } from "../acp/translate-inbound.js";

/**
 * Speak ACP to a child agent over stdio. Drop-in compatible with the
 * `{ run, cancel }` shape that TurnBroker (turn-broker.js:104) and
 * RunScheduler (run-scheduler.js:64) already expect.
 */
export class AcpRunner {
  constructor({ binary, args = [], cwd = process.cwd(), env, memoryRoot, onEvent } = {}) {
    if (!binary) throw new Error("AcpRunner: binary is required");
    this.binary = binary;
    this.args = args;
    this.cwd = cwd;
    this.env = env;
    this.memoryRoot = memoryRoot;
    this.onEvent = onEvent;
    this.child = null;
  }

  async run({ prompt, signal } = {}) {
    if (!prompt?.trim()) throw new Error("AcpRunner: prompt must be non-empty");

    const child = spawn(this.binary, this.args, {
      cwd: this.cwd,
      env: { ...process.env, ...(this.memoryRoot ? { HERMES_HOME: this.memoryRoot } : {}), ...(this.env ?? {}) },
      stdio: ["pipe", "pipe", "pipe"]
    });
    this.child = child;
    let stderr = "";
    child.stderr.setEncoding("utf8");
    child.stderr.on("data", (c) => { stderr += c; });

    const onEvent = this.onEvent;
    const conn = new ClientSideConnection(
      () => ({
        async sessionUpdate({ sessionId, update }) {
          if (!onEvent) return;
          const translated = acpUpdateToSpecifyrEvent(update);
          if (translated) onEvent(translated);
        },
        async requestPermission() {
          // Filled in Task 1.5
          return { outcome: { outcome: "selected", optionId: "reject_once" } };
        },
        async readTextFile() { throw new Error("fs/read_text_file not implemented yet (Task 1.4)"); },
        async writeTextFile() { throw new Error("fs/write_text_file not implemented yet (Task 1.4)"); }
      }),
      Writable.toWeb(child.stdin),
      Readable.toWeb(child.stdout)
    );

    const onAbort = () => { if (this.child && !this.child.killed) this.child.kill("SIGTERM"); };
    if (signal) {
      if (signal.aborted) onAbort();
      else signal.addEventListener("abort", onAbort, { once: true });
    }

    try {
      const init = await conn.initialize({
        protocolVersion: 1,
        clientCapabilities: { fs: { readTextFile: true, writeTextFile: true }, terminal: false }
      });
      const newSession = await conn.newSession({ cwd: this.cwd, mcpServers: [] });
      const sessionId = newSession.sessionId;
      const promptResult = await conn.prompt({
        sessionId,
        prompt: [{ type: "text", text: prompt }]
      });
      const exitCode = 0;
      child.kill();
      return {
        claudeSessionId: null,
        result: { type: "result", subtype: promptResult.stopReason === "end_turn" ? "success" : "error", result: "" },
        exitCode,
        stderr
      };
    } catch (err) {
      if (signal?.aborted) {
        const e = new Error("Aborted");
        e.aborted = true;
        throw e;
      }
      throw err;
    } finally {
      this.child = null;
    }
  }

  cancel() {
    if (this.child && !this.child.killed) this.child.kill("SIGTERM");
  }
}
```

// src/acp/translate-inbound.js
/**
 * Convert one ACP SessionUpdate into one specifyr-shaped onEvent payload:
 *   { type: "assistant", message: { role: "assistant", content: [...] } }
 * Returns null if the update has no analogue.
 */
export function acpUpdateToSpecifyrEvent(update) {
  if (!update || typeof update !== "object") return null;
  switch (update.sessionUpdate) {
    case "agent_message_chunk":
      if (update.content?.type === "text") {
        return {
          type: "assistant",

        };
      }
      return null;
    case "tool_call":
      return {
        type: "assistant",
        message: {
          role: "assistant",
          content: [{ type: "tool_use", id: update.toolCallId, name: update.title, input: update.rawInput ?? {} }]
        }
      };
    case "tool_call_update":
      return {
        type: "user",
        message: {
          role: "user",
          content: [{
            type: "tool_result",
            tool_use_id: update.toolCallId,
            content: textOf(update.content),
            is_error: update.status === "failed"
          }]
        }
      };
    default:
      return null;
  }
}

function textOf(content) {
  if (!Array.isArray(content)) return "";
  for (const item of content) {
    if (item?.content?.type === "text") return item.content.text;
  }
  return "";
}

// src/acp/fs-handlers.js
import path from "node:path";
import { readFile, writeFile, mkdir } from "node:fs/promises";

export function makeFsHandlers({ cwd }) {
  if (!cwd || !path.isAbsolute(cwd)) {
    throw new Error("makeFsHandlers: absolute cwd required");
  }
  const root = path.resolve(cwd);

  function check(p) {
    if (typeof p !== "string" || p.length === 0) throw new Error("path required");
    if (!path.isAbsolute(p)) throw new Error("path must be absolute");
    const resolved = path.resolve(p);
    const rel = path.relative(root, resolved);
    if (rel.startsWith("..") || path.isAbsolute(rel)) {
      throw new Error(`path outside session cwd: ${resolved}`);
    }
    return resolved;
  }

  return {
    async readTextFile({ path: p, line, limit }) {
      const safe = check(p);
      const content = await readFile(safe, "utf8");
      if (line == null && limit == null) return { content };
      const lines = content.split("\n");
      const startIdx = Math.max(0, (line ?? 1) - 1);
      const sliced = limit != null ? lines.slice(startIdx, startIdx + limit) : lines.slice(startIdx);
      return { content: sliced.join("\n") };
    },
    async writeTextFile({ path: p, content }) {
      const safe = check(p);
      await mkdir(path.dirname(safe), { recursive: true });
      await writeFile(safe, content, "utf8");
      return {};
    }
  };
}
```

async function resolveSlugFromCwd(projectRoot, cwd) {

import { specifyrEventToAcpUpdates } from "./translate.js";

export function createSpecifyrAcpAgent({ client, projectRoot, turnBroker }) {
  return {
    // ...existing handlers...
    async prompt({ sessionId, prompt }) {
      const { slug, stepId, sid } = decodeSessionId(sessionId);
      const text = prompt.filter((b) => b.type === "text").map((b) => b.text).join("\n");
      const cwd = projectRoot;
      const emitter = turnBroker.emitterFor(slug, stepId, sid);
      let stopReason = "end_turn";

      const done = new Promise((resolve) => {
        const onEvent = async (entry) => {
          if (entry.event === "turn_failed") {
            stopReason = "refusal";
          }
          for (const update of specifyrEventToAcpUpdates(entry)) {
            try { await client.sessionUpdate({ sessionId, update }); }
            catch (err) { /* client gone — keep persisting via TurnBroker */ }
          }
        };
        const onEnded = () => {
          emitter.off("event", onEvent);
          emitter.off("ended", onEnded);
          resolve();
        };
        emitter.on("event", onEvent);
        emitter.on("ended", onEnded);
      });

      await turnBroker.startTurn({ slug, stepId, sid, prompt: text, cwd });
      await done;
      return { stopReason };
    },
    async cancel({ sessionId }) {
      const { slug, stepId, sid } = decodeSessionId(sessionId);
      turnBroker.cancel?.(slug, stepId, sid);
    }
  };
}

import { AgentSideConnection } from "@agentclientprotocol/sdk";

import { TurnBroker } from "../src/core/turn-broker.js";

const projectRoot = process.cwd();
const appConfig = await loadAppConfig(projectRoot);
const sessionStore = new SessionStore(projectRoot);

function pickRunnerFactory() {
  for (const name of appConfig.runner.fallbackChain) {
    if (name.startsWith("acp:")) {
      const cfg = appConfig.acp?.[name.slice(4)];
      if (cfg?.binary) return ({ cwd, onEvent }) => new AcpRunner({ binary: cfg.binary, args: cfg.args, cwd, onEvent });

    if (name === "claude") {
      return ({ cwd, onEvent }) => new ClaudeCodeRunner({ binary: appConfig.claude.binary, cwd, onEvent });
    }
  }
  throw new Error("no runner available");
}

const turnBroker = new TurnBroker({ sessionStore, runnerFactory: pickRunnerFactory() });

const stdin = Readable.toWeb(process.stdin);
const stdout = Writable.toWeb(process.stdout);

new AgentSideConnection(
  (client) => createSpecifyrAcpAgent({ client, projectRoot, turnBroker }),

  stdin

// src/acp/approval-transport.js
export class AcpApprovalTransport {
  constructor({ client }) {
    this.client = client;
    this.bindings = new Map(); // slug -> sessionId
  }

  bindSession({ slug }, sessionId) {
    this.bindings.set(slug, sessionId);
  }

  unbind(slug) { this.bindings.delete(slug); }

  /**
   * Called by CapabilityApprovalService when an approval is needed.
   * Returns "approved" | "denied".
   */
  async notify({ slug, capability, requestPayload }) {
    const sessionId = this.bindings.get(slug);
    if (!sessionId) return "denied"; // no client connected → safe default
    const toolCall = requestPayload?.toolCall ?? { title: capability, toolCallId: `cap-${Date.now()}` };
    const result = await this.client.requestPermission({
      sessionId,
      toolCall,
      options: [
        { optionId: "allow_once", name: `Allow ${capability} once`, kind: "allow_once" },
        { optionId: "allow_always", name: `Allow ${capability} always`, kind: "allow_always" },
        { optionId: "reject_once", name: "Deny", kind: "reject_once" },
        { optionId: "reject_always", name: "Deny always", kind: "reject_always" }
      ]
    });
    if (result.outcome?.outcome === "selected" && result.outcome.optionId.startsWith("allow")) {
      return "approved";
    }
    return "denied";
  }
}

## Notes for the executing engineer

- **SDK API drift:** the SDK is at version 0.21.0 and the protocol is at version 1.
